Código Java

Inserción normal (al final) e inserción con prioridad


// === INSERCIÓN NORMAL ===
void insertarNormal(Nodo[] cola,
        int turno, String tipo,
        String nombre) {
    Nodo nuevo = new Nodo(
        turno, tipo, 0, nombre);
    if (cola[0] == null) {
        cola[0] = nuevo;
        return;
    }
    Nodo actual = cola[0];
    while (actual.siguiente
           != null)
        actual = actual.siguiente;
    actual.siguiente = nuevo;
}
      

// === INSERCIÓN PRIORITARIA ===
void insertarPrioridad(Nodo[] cola,
        int turno, String tipo,
        String nombre) {
    Nodo nuevo = new Nodo(
        turno, tipo, 1, nombre);
    if (cola[0] == null ||
        cola[0].prioridad < 1) {
        nuevo.siguiente = cola[0];
        cola[0] = nuevo;
        return;
    }
    Nodo actual = cola[0];
    while (actual.siguiente != null
           && actual.siguiente
               .prioridad >= 1)
        actual = actual.siguiente;
    nuevo.siguiente = actual.siguiente;
    actual.siguiente = nuevo;
}
