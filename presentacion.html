📝 Estructura del Nodo

struct Nodo
●turno : int ●tipo : char[] ●prioridad : int ●nombre : char[] →siguiente : Nodo*
prioridad: 1 = alta, 0 = normal

📌 Características

  • Inserción por orden de llegada
  • Los turnos prioritarios se colocan al inicio
  • Los clientes con misma prioridad respetan FIFO
  • Estructura dinámica: crece y se reduce según demanda

🔢 Insertar con Prioridad

Ejemplo: un adulto mayor ingresa y se coloca antes que los turnos normales 


typedef struct Nodo {
    int turno;
    char tipo[30];
    int prioridad;   // 1 = alta, 0 = normal
    char nombre[50];
    struct Nodo* siguiente;
} Nodo;

void insertarConPrioridad(Nodo** cola, int turno, char* tipo,
                           int prioridad, char* nombre) {
    Nodo* nuevo = (Nodo*)malloc(sizeof(Nodo));
    nuevo->turno = turno;
    nuevo->prioridad = prioridad;
    strcpy(nuevo->nombre, nombre);
    strcpy(nuevo->tipo, tipo);

    // Si la cola está vacía o el nuevo tiene prioridad máxima
    if (*cola == NULL || prioridad > (*cola)->prioridad) {
        nuevo->siguiente = *cola;
        *cola = nuevo;
        return;
    }

    // Buscar la posición correcta respetando FIFO
    Nodo* actual = *cola;
    while (actual->siguiente != NULL &&
           actual->siguiente->prioridad >= prioridad) {
        actual = actual->siguiente;
    }
    nuevo->siguiente = actual->siguiente;
    actual->siguiente = nuevo;
}

// Uso: insertar adulto mayor con prioridad 1
insertarConPrioridad(&cola, 107, "Caja", 1, "Don José");
